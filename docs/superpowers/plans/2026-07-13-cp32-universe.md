# CP 3.2 — Universbygger: Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Reconstruct historical S&P 500 membership so `universe(D)` returns the companies investable on date D — including ones later delisted — with no survivorship bias.

**Architecture:** Pure functions in `backend/pipeline/sp500.py` — `membership_on` (backward reconstruction, the correctness oracle) and `build_intervals` (forward replay → dateranges + companies + anomalies). Persistence upserts `companies` and rebuilds `index_constituents` in one transaction; `universe(conn, D)` is a `membership @> D` query. Two independent computations are cross-checked.

**Tech Stack:** Python (stdlib datetime), psycopg, FMPClient (CP 3.1), pytest.

## Global Constraints

- `uv run pytest backend/tests/ -q` from repo root; deps via `uv add`.
- FMP change-log fields (verified): `date` (ISO), `symbol`+`addedSecurity` (added), `removedTicker`+`removedSecurity` (removed, empty on pure add), `reason`. Today's list fields: `symbol`, `name`, `dateFirstAdded`.
- `daterange` is half-open `[start, end)`; `membership @> D` means D in interval. `end=None` → `daterange(start, NULL)` = `[start, infinity)`.
- `index_constituents.index_code = 'SP500'`. Idempotent persist = delete-all-SP500 + insert in one transaction.
- Anomalies are logged and returned, never raised (03 principle 5).
- CP 3.2 sets only ticker/name/is_delisted/delisted_date on companies (sector etc. = S2).
- Commit per task; push at end.

```python
"""S&P 500 historical membership reconstruction (CP 3.2, 03 S1).

Two independent computations kept deliberately: membership_on (backward
reconstruction from today's set — the correctness oracle) and build_intervals
(forward replay producing the persisted dateranges). A cross-check test
asserts they agree on historical dates. Pure functions, no I/O.
"""

from __future__ import annotations

import datetime as dt
from dataclasses import dataclass, field


def _parse(d: str) -> dt.date:
    return dt.date.fromisoformat(d)


def membership_on(
    today_symbols: set[str], change_log: list[dict], on: dt.date
) -> set[str]:
    """Set of tickers in the index on `on`, by undoing every change after it."""
    members = set(today_symbols)
    for row in sorted(change_log, key=lambda r: r["date"], reverse=True):
        if _parse(row["date"]) <= on:
            break
        added = row["symbol"].strip()
        removed = row["removedTicker"].strip()
        if added:
            members.discard(added)
        if removed:
            members.add(removed)
    return members
```

### Task 2: `build_intervals` — forward replay → dateranges + companies + anomalies

**Files:**
- Modify: `backend/pipeline/sp500.py`

**Interfaces:**
- Consumes: `membership_on` (for the cross-check test only).
- Produces:
  - `@dataclass Constituent(ticker: str, name: str, start: date, end: date | None)`
  - `@dataclass Company(ticker: str, name: str, is_delisted: bool, delisted_date: date | None)`
  - `@dataclass Anomaly(kind: str, date: str, ticker: str, detail: str)`
  - `@dataclass UniverseBuild(companies: list[Company], constituents: list[Constituent], anomalies: list[Anomaly])`
  - `build_intervals(today: list[dict], change_log: list[dict]) -> UniverseBuild`

Algorithm: collect per-ticker events `(date, kind, name)` where kind∈{"add","remove"} from the log (`symbol`/`addedSecurity` → add; `removedTicker`/`removedSecurity` → remove). Sort each ticker's events by date. State-machine walk producing disjoint intervals: on "add" while out → open interval at date; on "remove" while in → close [start,date); on "add" while in or "remove" while out → Anomaly, skip. Reconcile with today (`{r["symbol"] for r in today}`): if ticker in today's set and last interval open → end=None (current); if in today but ended closed → Anomaly("closed_but_current"); if not in today and ends open → close it? No: Anomaly("open_but_absent") and drop the trailing open interval (don't claim current membership). Founding members present in today with a single open interval → `[first_add, None]`. Names: prefer today's `name` for current members, else the most recent event name. is_delisted = ticker not in today's set; delisted_date = last remove date for delisted.

```python
@dataclass(frozen=True)
class Constituent:
    ticker: str
    name: str
    start: dt.date
    end: dt.date | None  # None = [start, infinity) (current member)


@dataclass(frozen=True)
class Company:
    ticker: str
    name: str
    is_delisted: bool
    delisted_date: dt.date | None


@dataclass(frozen=True)
class Anomaly:
    kind: str
    date: str
    ticker: str
    detail: str


@dataclass
class UniverseBuild:
    companies: list[Company] = field(default_factory=list)
    constituents: list[Constituent] = field(default_factory=list)
    anomalies: list[Anomaly] = field(default_factory=list)


def build_intervals(today: list[dict], change_log: list[dict]) -> UniverseBuild:
    today_syms = {r["symbol"] for r in today}
    today_name = {r["symbol"]: r.get("name", r["symbol"]) for r in today}

    # Gather per-ticker events. Each event carries the security name seen with it.
    events: dict[str, list[tuple[dt.date, str, str]]] = {}
    for row in change_log:
        d = _parse(row["date"])
        added = row["symbol"].strip()
        removed = row["removedTicker"].strip()
        if added:
            events.setdefault(added, []).append(
                (d, "add", row.get("addedSecurity", "").strip())
            )
        if removed:
            events.setdefault(removed, []).append(
                (d, "remove", row.get("removedSecurity", "").strip())
            )

    build = UniverseBuild()
    for ticker, evs in events.items():
        evs.sort(key=lambda e: e[0])
        name = today_name.get(ticker) or next(
            (n for _, _, n in reversed(evs) if n), ticker
        )
        open_start: dt.date | None = None
        last_remove: dt.date | None = None
        for d, kind, _n in evs:
            if kind == "add":
                if open_start is None:
                    open_start = d
                else:
                    build.anomalies.append(
                        Anomaly("double_add", d.isoformat(), ticker,
                                "add while already a member")
                    )
            else:  # remove
                if open_start is not None:
                    build.constituents.append(
                        Constituent(ticker, name, open_start, d)
                    )
                    open_start = None
                    last_remove = d
                else:
                    build.anomalies.append(
                        Anomaly("remove_without_add", d.isoformat(), ticker,
                                "remove while not a member")
                    )
        in_today = ticker in today_syms
        if open_start is not None:
            if in_today:
                build.constituents.append(
                    Constituent(ticker, name, open_start, None)
                )
            else:
                build.anomalies.append(
                    Anomaly("open_but_absent", open_start.isoformat(), ticker,
                            "log leaves ticker current but absent from today's list")
                )
        elif in_today:
            build.anomalies.append(
                Anomaly("closed_but_current", "", ticker,
                        "today's list has ticker but log ended removed")
            )
        build.companies.append(
            Company(
                ticker=ticker,
                name=name,
                is_delisted=not in_today,
                delisted_date=None if in_today else last_remove,
            )
        )

    # Current members with no events at all (edge: founding, never changed) —
    # none expected from this log, but add them from today's list defensively.
    for row in today:
        if row["symbol"] not in events:
            build.companies.append(
                Company(row["symbol"], today_name[row["symbol"]], False, None)
            )
            # No event dates -> cannot bound the start; use dateFirstAdded.
            start = row.get("dateFirstAdded")
            build.constituents.append(
                Constituent(
                    row["symbol"], today_name[row["symbol"]],
                    _parse(start) if start else dt.date(1957, 3, 4), None,
                )
            )
    return build
```

```python
"""Persist reconstructed S&P 500 membership (CP 3.2, 03 S1 / 02 §4)."""

from __future__ import annotations

import datetime as dt

from psycopg.types.range import Range

from backend.pipeline.sp500 import UniverseBuild

INDEX = "SP500"


def _company_id(cur, ticker: str, name: str, is_delisted: bool,
                delisted_date: dt.date | None) -> int:
    """Upsert on the natural key (ticker, name); name disambiguates reuse."""
    row = cur.execute(
        "select id from companies where ticker=%s and name=%s", (ticker, name)
    ).fetchone()
    if row:
        cur.execute(
            "update companies set is_delisted=%s, delisted_date=%s where id=%s",
            (is_delisted, delisted_date, row[0]),
        )
        return row[0]
    return cur.execute(
        """insert into companies (ticker, name, is_delisted, delisted_date)
           values (%s, %s, %s, %s) returning id""",
        (ticker, name, is_delisted, delisted_date),
    ).fetchone()[0]


def persist_universe(conn, build: UniverseBuild) -> dict:
    with conn.cursor() as cur:
        ids: dict[tuple[str, str], int] = {}
        for c in build.companies:
            ids[(c.ticker, c.name)] = _company_id(
                cur, c.ticker, c.name, c.is_delisted, c.delisted_date
            )
        cur.execute("delete from index_constituents where index_code=%s", (INDEX,))
        for con in build.constituents:
            cid = ids[(con.ticker, con.name)]
            # Half-open [start, end); end=None -> [start, infinity).
            rng = Range(con.start, con.end, bounds="[)")
            cur.execute(
                """insert into index_constituents (index_code, company_id, membership)
                   values (%s, %s, %s)""",
                (INDEX, cid, rng),
            )
    conn.commit()
    return {"companies": len(build.companies),
            "constituents": len(build.constituents),
            "anomalies": len(build.anomalies)}


def universe(conn, on: dt.date) -> set[int]:
    rows = conn.execute(
        """select company_id from index_constituents
           where index_code=%s and membership @> %s""",
        (INDEX, on),
    ).fetchall()
    return {r[0] for r in rows}
```
